add_namespace = faction_alignment

# Under Threat - Seek Protection
country_event = {
	id = faction_alignment.1
	title = faction_alignment.1.t
	desc = faction_alignment.1.d
	picture = GFX_report_event_generic_sign_treaty2
	
	trigger = {
		is_subject = no
		NOT = { is_in_faction = yes }
		NOT = { has_war = yes }
		any_neighbor_country = {
			NOT = { has_government = ROOT }
			OR = {
				has_war_support > 0.5
				has_idea = war_economy
				has_idea = partial_economic_mobilisation
			}
			NOT = { is_in_faction_with = ROOT }
			strength_ratio = {
				tag = ROOT
				ratio > 1.2
			}
		}
		any_other_country = {
			is_faction_leader = yes
			has_government = ROOT
			NOT = { has_war_with = ROOT }
		}
	}
	
	mean_time_to_happen = {
		days = 120
		
		modifier = {
			factor = 0.7
			any_neighbor_country = {
				has_wargoal_against = ROOT
			}
		}
		
		modifier = {
			factor = 0.5
			any_neighbor_country = {
				has_war_with = ROOT
			}
		}
		
		modifier = {
			factor = 0.8
			has_defensive_war = yes
		}
	}
	
	# Seek their protection
	option = {
		name = faction_alignment.1.a
		ai_chance = {
			factor = 80
			
			modifier = {
				factor = 1.5
				any_neighbor_country = {
					has_wargoal_against = ROOT
				}
			}
			
			modifier = {
				factor = 2.0
				has_defensive_war = yes
			}
		}
		
		random_other_country = {
			limit = {
				is_faction_leader = yes
				has_government = ROOT
				NOT = { has_war_with = ROOT }
			}
			country_event = { id = faction_alignment.2 days = 1 }
			save_event_target_as = potential_faction_leader
		}
		
		add_political_power = -50
	}
	
	# We stand alone
	option = {
		name = faction_alignment.1.b
		ai_chance = {
			factor = 20
		}
		
		add_stability = 0.03
		add_war_support = 0.05
	}
}

# Request to Join Faction
country_event = {
	id = faction_alignment.2
	title = faction_alignment.2.t
	desc = faction_alignment.2.d
	picture = GFX_report_event_generic_sign_treaty1
	
	is_triggered_only = yes
	
	# Accept them
	option = {
		name = faction_alignment.2.a
		ai_chance = {
			factor = 70
			
			modifier = {
				factor = 1.5
				FROM = {
					any_neighbor_country = {
						has_wargoal_against = FROM
					}
				}
			}
			
			modifier = {
				factor = 2.0
				FROM = {
					has_defensive_war = yes
				}
			}
			
			modifier = {
				factor = 0.5
				FROM = {
					surrender_progress > 0.5
				}
			}
		}
		
		add_to_faction = FROM
		FROM = {
			country_event = { id = faction_alignment.3 days = 1 }
		}
	}
	
	# Decline
	option = {
		name = faction_alignment.2.b
		ai_chance = {
			factor = 30
		}
		
		FROM = {
			country_event = { id = faction_alignment.4 days = 1 }
		}
	}
}
